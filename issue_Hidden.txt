	 public String viewTerminalOnboardAanipayDetails() {
    	
    /*	if((aaniPayMerchantDetailsBean.getMerchantId() == null || aaniPayMerchantDetailsBean.getMerchantId().isEmpty())
    		&& (httpServletRequest.getParameter("merchantId") != null || !httpServletRequest.getParameter("merchantId").isEmpty())) {
    		System.out.println(httpServletRequest.getParameter("merchantId"));
    		aaniPayMerchantDetailsBean.setMerchantId(httpServletRequest.getParameter("merchantId"));
    	}*/
		
    	if (aaniPayMerchantDetailsBean == null || aaniPayMerchantDetailsBean.getSelectMerchantAanipayApi() == null || 
    	    aaniPayMerchantDetailsBean.getMerchantId() == null || aaniPayMerchantDetailsBean.getStoreId() == null) {
			addActionError(getText("error.merchant.details.invalid"));
			return ERROR;
		}
		
		setStoreDetailsList(merchantOnboardAanipayService
				            .getaaniPayStoreDetails(aaniPayMerchantDetailsBean.getSelectMerchantAanipayApi(), 
				            aaniPayMerchantDetailsBean.getMerchantId(),
				            sessionBean));
		try {

			String storeStatus = merchantOnboardAanipayService
					.validateStorStatusBasedOnSid(aaniPayMerchantDetailsBean.getStoreId(), sessionBean.getInstCode());
			if (storeStatus == null) {
				addActionError("No store status found. Please check the Store ID.");
				return ERROR;
			} else {
				terminalDetailsList = merchantOnboardAanipayService
						        .getaaniPayTerminalDetails(storeStatus, aaniPayMerchantDetailsBean.getStoreId(), sessionBean);

			}

			if (terminalDetailsList == null || terminalDetailsList.isEmpty()) {
				addActionError("No Store Found ");
				return ERROR;
			}

		} catch (Exception e) {
			log.error("Error fetching store details", e);
			addActionError("Error while  fetching store details   ");
			return ERROR;
		}

		return SUCCESS;
	}
