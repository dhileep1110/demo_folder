package com.fss.maps.common.util;

import java.net.InetAddress;
import java.time.LocalDateTime;
import java.time.format.DateTimeFormatter;
import java.util.LinkedHashMap;
import java.util.Map;
import java.util.UUID;

import com.fasterxml.jackson.databind.ObjectMapper;

public class EmailAPIHeaderBuilder {
	
    private static final String CHANNEL_ID = "FSS";
    private static final String SERVICE_ID = "9100";
    private static final String REGION_ID = "UAE";
    private static final String FINANCIAL_ID = "DIB";
    private static final String TIMEZONE = "GMT+04:00";
    private static final String PRIVATE_KEY_PATH = "/app/MAPS/DIB/MAPSCore/FSS_UAT_PRIVATE.key";

    /**
     * Builds the full DIB API header dynamically
     */
    public static Map<String, String> buildHeader(String requestBody) {
        Map<String, String> header = new LinkedHashMap<>();
        try {
            //Dynamic IP Address
            String ip = InetAddress.getLocalHost().getHostAddress();

            //Unique Reference ID
            String refNo = UUID.randomUUID().toString().replace("-", "").substring(0, 16);

            //Current TimeStamp
            String timestamp = LocalDateTime.now().format(DateTimeFormatter.ofPattern("yyyy-MM-dd'T'HH:mm:ss:SSS"));

            //Hash PayLoad using HMAC-SHA256
            String hashPayload = EmailApiSecurityUtil.calcHmacSha256(CHANNEL_ID, requestBody);

            //JWS Token Generation
            String jws = EmailApiSecurityUtil.createJwsToken(CHANNEL_ID, refNo, PRIVATE_KEY_PATH);

            //Dynamic User-Agent info (OS + Java version + hostName)
            String systemUserAgent = generateSystemUserAgent();

            //Build header
            header.put("channelId", CHANNEL_ID);
            header.put("client-ip-address", ip);
            header.put("financialId", FINANCIAL_ID);
            header.put("hashPayload", hashPayload);
            header.put("hashHeader", null);
            header.put("jws", jws);
            header.put("regionId", REGION_ID);
            header.put("requestTimestamp", timestamp);
            header.put("requestTimeZone", TIMEZONE);
            header.put("requestUniqueRefNo", refNo);
            header.put("user-agent", systemUserAgent);
            header.put("serviceId", SERVICE_ID);

        } catch (Exception e) {
            e.printStackTrace();
        }
        return header;
    }

    /**
     * Builds a readable system user-agent string like:
     * "Windows 10 | Java 17 | Host: DESKTOP-1234"
     */
    private static String generateSystemUserAgent() {
        String osName = System.getProperty("os.name");
        String osVersion = System.getProperty("os.version");
        String javaVersion = System.getProperty("java.version");
        String hostName = "UNKNOWN";

        try {
            hostName = InetAddress.getLocalHost().getHostName();
        } catch (Exception ignored) { }

        return String.format("%s %s | Java %s | Host: %s", osName, osVersion, javaVersion, hostName);
    }


}
